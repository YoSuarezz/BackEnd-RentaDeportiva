-- Crear el esquema rentadeportiva
CREATE SCHEMA rentadeportiva;

-- Crear la tabla Pais
CREATE TABLE rentadeportiva.pais (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL
);

-- Crear la tabla Departamento
CREATE TABLE rentadeportiva.departamento (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    pais_id INT NOT NULL REFERENCES rentadeportiva.pais(id)
);

-- Crear la tabla Ciudad
CREATE TABLE rentadeportiva.ciudad (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    departamento_id INT NOT NULL REFERENCES rentadeportiva.departamento(id)
);

-- Crear la tabla Responsable
CREATE TABLE rentadeportiva.responsable (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    apellido VARCHAR(255) NOT NULL,
    correo VARCHAR(255) NOT NULL,
    telefono INT NOT NULL,
    estado VARCHAR(255) NOT NULL
);

-- Crear la tabla TipoDeDocumento
CREATE TABLE rentadeportiva.tipodedocumento (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL
);

-- Crear la tabla Validacion
CREATE TABLE rentadeportiva.validacion (
    id SERIAL PRIMARY KEY,
    correo VARCHAR(255) NOT NULL,
    contrasena VARCHAR(255) NOT NULL
);

-- Crear la tabla UnidadDeportiva
CREATE TABLE rentadeportiva.unidaddeportiva (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    numero_documento INT NOT NULL,
    direccion VARCHAR(255) NOT NULL,
    telefono INT NOT NULL,
    ciudad_id INT NOT NULL REFERENCES rentadeportiva.ciudad(id),
    correo_id INT NOT NULL REFERENCES rentadeportiva.validacion(id),
    responsable_id INT NOT NULL REFERENCES rentadeportiva.responsable(id),
    tipo_documento_id INT NOT NULL REFERENCES rentadeportiva.tipodedocumento(id)
);
